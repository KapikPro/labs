void input(int& n, double*& x)
{
    FILE* f;
    if ((f = fopen("fin.txt", "r")) == NULL)
    {
        cout << "ошибка открытия файла\n";
        exit(0);
    }
    fscanf(f, "%d", &n);
    x = new double[n];
    for (int i = 0; i < n; i++)
    {
        fscanf(f, "%lf", (x + i));
    }
    fclose(f);
}

int counting(int a,int b,double *x)
{
    int count = 0;
    if (b - a != 0 && b-a!=1)
    {
        count+=counting(a, (a+b) / 2, x);
        count+=counting((a+b) / 2, b, x);
    }
    else
        if (b - a == 1)
        {
           count+= counting(a, a, x);
           count+= counting(b, b, x);
        }
        else
        if (sin((*(x + a)) / 2) < 0)
            count++;
    return count;
}
